── com.example.decoratorapi.component.mapper.ComponentMapper  @Component
  constructor inyecta InMemoryDecoratorStore
  toResponse(ComponentRecord record):
    - Itera record.getAppliedDecoratorTypes()
    - Por cada tipo: decoratorStore.findByType(tipo).map(DecoratorDefinition::getAdditionalCost).orElse(BigDecimal.ZERO)
    - Crea DecoratorCostItem(tipo, costo) por cada uno
    - totalAdded = lista.stream().map(DecoratorCostItem::getCost).reduce(BigDecimal.ZERO, BigDecimal::add)
    - Construye PriceBreakdownResponse(record.getBasePrice(), lista, totalAdded, record.getFinalPrice())
    - Retorna ComponentResponse con todos los campos del record + priceBreakdown
  toResponseList(List<ComponentRecord> records) → records.stream().map(this::toResponse).toList()

── com.example.decoratorapi.component.service.ComponentService  @Service
  constructor inyecta: InMemoryComponentStore, InMemoryDecoratorStore, DecoratorFactory, ComponentMapper

  createComponent(CreateComponentRequest req):
    Component base = new BaseProductComponent(UUID.randomUUID().toString(), req.getName(), req.getDescription(), req.getBasePrice());
    return componentMapper.toResponse(componentStore.save(ComponentRecord.fromDomain(base)));

  getComponent(String id):
    ComponentRecord r = componentStore.findById(id).orElseThrow(() -> new ResourceNotFoundException("Component not found with id: " + id));
    return componentMapper.toResponse(r);

  getAllComponents():
    return componentMapper.toResponseList(componentStore.findAll());

  applyDecorator(String componentId, ApplyDecoratorRequest req):
    1. Busca record → lanza 404 si no existe
    2. Si !decoratorStore.exists(req.getDecoratorType()) → throw new InvalidDecoratorException("Unknown decorator type: " + req.getDecoratorType())
    3. Component decorated = decoratorFactory.apply(rebuildChain(record), req.getDecoratorType())
    4. ComponentRecord updated = ComponentRecord.fromDomain(decorated); updated.setCreatedAt(record.getCreatedAt());
    5. return componentMapper.toResponse(componentStore.save(updated));

  removeDecorator(String componentId, String decoratorType):
    1. Busca record → lanza 404 si no existe
    2. String typeUpper = decoratorType.toUpperCase();
       Si !record.getAppliedDecoratorTypes().contains(typeUpper) → throw new InvalidDecoratorException("Decorator " + typeUpper + " not found on component")
    3. List<String> remaining = new ArrayList<>(record.getAppliedDecoratorTypes()); remaining.remove(typeUpper);
    4. Component rebuilt = decoratorFactory.applyAll(new BaseProductComponent(record.getId(), record.getName(), record.getName(), record.getBasePrice()), remaining);
    5. ComponentRecord updated = ComponentRecord.fromDomain(rebuilt); updated.setCreatedAt(record.getCreatedAt());
    6. return componentMapper.toResponse(componentStore.save(updated));

  deleteComponent(String id): componentStore.deleteById(id);

  simulateComposition(CreateComponentRequest req, List<String> types):
    Component base = new BaseProductComponent(UUID.randomUUID().toString(), req.getName(), req.getDescription(), req.getBasePrice());
    Component composed = decoratorFactory.applyAll(base, types);
    return componentMapper.toResponse(ComponentRecord.fromDomain(composed));
    // NO guardar en el store

  MÉTODO PRIVADO rebuildChain(ComponentRecord record):
    Component base = new BaseProductComponent(record.getId(), record.getName(), record.getName(), record.getBasePrice());
    return decoratorFactory.applyAll(base, record.getAppliedDecoratorTypes());
